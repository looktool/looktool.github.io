import urllib.request

def load_data():
    url = r"http://www.baidu.com"
    # get 请求
    response = urllib.request.urlopen(url)
    print(response)
    # 读取内容
    data = response.read()
    print(data)
    # 解码
    str_data = data.decode("utf-8")
    print(str_data)
    # 写入文件
    with open("baidu.html","w",encoding="utf-8")as f:
        f.write(str_data)

        
if __name__ == "__main__":
    load_data()
